Sounds

[snd]
fname		= file.wav
autorun		= on/off
b3d			= on/off
loopmode	= on/off
loopcount	= раз
volume		= -10000..0
mindist		= 1.0 м
maxdist		= 100.0 м
pos			= 95.0, 2.0, 123.0
random		= с какой максимальной периодичностью запускать - мсек
dir			= вектор направления
in_angle	= 0..360 градусов
out_angle	= 0..360 градусов
out_volume	= -10000..0


fname		= имя ф-ла
b3d			= on/off - звук 3d или нет
random		= интервал между запусками (мсек)
loopmode	= on/off - зациклен или нет
loopcount	= если он зациклен то сколько раз может проиграть если 0 - бесконечно
autorun		= on/off - после загрузки автоматически сам запускается
volume		= -10000..0 - громкость (0-максимум)
pos			= x,y,z - позиция
dir			= x,y,z - направление если в виде конуса
mindist		= мин. дистанция от которой громкость уменьшается
maxdist		= макс. дистанция до которой слышно звук
in_angle	= внутренний угол (радиан)
out_angle	= внешний угол (радиан)
out_volume	= -10000..0 - громкость за внешним углом

fname		= ????

loopcount	= 0
